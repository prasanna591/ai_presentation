üìö Machine Learning: The Core of Modern AI

Types of Machine Learning:

  • Supervised Learning: Learning from labeled examples (Classification, Regression)
  • Unsupervised Learning: Finding patterns in unlabeled data (Clustering, Dimensionality Reduction)
  • Reinforcement Learning: Learning through trial and error with rewards

Key Concepts:

  • Training Data: Examples used to teach the algorithm
  • Features: Input variables used to make predictions
  • Model: The mathematical representation learned from data
  • Prediction: Output generated by the trained model

import numpy as np
import matplotlib.pyplot as plt
from sklearn.linear_model import LinearRegression
from sklearn.model_selection import train_test_split
from sklearn.metrics import mean_squared_error, r2_score

class HousePricePredictor:
    def __init__(self):
        self.model = LinearRegression()
        self.is_trained = False
        
    def generate_data(self, n_samples=100):
        """Generate synthetic house data"""
        np.random.seed(42)
        
        # Features: size (sq ft), location multiplier
        sizes = np.random.normal(2000, 800, n_samples)
        sizes = np.clip(sizes, 500, 5000)  # Realistic range
        
        location_multipliers = np.random.choice([0.5, 0.7, 1.0], n_samples, 
                                              p=[0.3, 0.4, 0.3])  # Rural, Suburban, Urban
        
        # Price formula with some noise
        base_price = 50000
        price_per_sqft = 150
        noise = np.random.normal(0, 20000, n_samples)
        
        prices = base_price + (sizes * price_per_sqft * location_multipliers) + noise
        prices = np.clip(prices, 50000, 1000000)  # Realistic price range
        
        self.X = np.column_stack([sizes, location_multipliers])
        self.y = prices
        
        return self.X, self.y
    
    def train(self):
        """Train the AI model"""
        if self.X is None or self.y is None:
            raise ValueError("Please generate data first!")
            
        # Split data into training and testing sets
        X_train, X_test, y_train, y_test = train_test_split(
            self.X, self.y, test_size=0.2, random_state=42
        )
        
        # Train the model
        self.model.fit(X_train, y_train)
        
        # Evaluate performance
        y_pred = self.model.predict(X_test)
        mse = mean_squared_error(y_test, y_pred)
        r2 = r2_score(y_test, y_pred)
        
        self.is_trained = True
        return mse, r2, X_test, y_test, y_pred
    
    def predict(self, size, location_multiplier):
        """Predict house price for given features"""
        if not self.is_trained:
            raise ValueError("Model must be trained first!")
            
        features = np.array([[size, location_multiplier]])
        prediction = self.model.predict(features)[0]
        return prediction
    
    def visualize_data(self):
        """Create visualization of the data"""
        if self.X is None:
            return None
            
        plt.figure(figsize=(10, 6))
        
        # Color code by location
        colors = ['red', 'orange', 'green']
        location_labels = ['Rural', 'Suburban', 'Urban']
        
        for i, multiplier in enumerate([0.5, 0.7, 1.0]):
            mask = self.X[:, 1] == multiplier
            plt.scatter(self.X[mask, 0], self.y[mask], 
                       c=colors[i], label=location_labels[i], alpha=0.7)
        
        plt.xlabel('House Size (sq ft)')
        plt.ylabel('Price ($)')
        plt.title('House Prices by Size and Location')
        plt.legend()
        plt.grid(True, alpha=0.3)
        return plt

# Example usage:
if __name__ == "__main__":
    # Create predictor instance
    predictor = HousePricePredictor()
    
    # Generate synthetic data
    X, y = predictor.generate_data(100)
    print(f"Generated {len(X)} house samples")
    
    # Train the model
    mse, r2, X_test, y_test, y_pred = predictor.train()
    print(f"Model trained! R¬≤ Score: {r2:.3f}")
    
    # Make predictions
    predicted_price = predictor.predict(2000, 1.0)  # 2000 sq ft urban house
    print(f"Predicted price for 2000 sq ft urban house: ${predicted_price:,.0f}")
    
    # Visualize results
    predictor.visualize_data()
    plt.show()
